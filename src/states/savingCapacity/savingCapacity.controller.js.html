'use strict';
 
angular.module('bpce.lea')
    .controller('SavingCapacityCtrl', function ($scope, $window, STATES, savingCapacity, savingCapacityService, savingCapacityBackend, dialog) {
        var savingCapacityCtrl = this;
 
        Object.defineProperties(savingCapacity, {
            total                   : {
                get: function () {
                    return savingCapacityService.getTotal(this);
                }
            },
            annualIncomeTaxesMonthly: {
                get: function () {
                    return savingCapacityService.getAnnualValueMonthly(this.annualIncomeTaxes);
                }
            }
        });
 
        ///////////////////////////////////////////////////////////
        //
        //      SCOPE DEFINITIONS
        //
 
        savingCapacityCtrl.savingCapacity = savingCapacity;
 
        savingCapacityCtrl.validate = validate;
        savingCapacityCtrl.showInternalSavingsDetails = showInternalSavingsDetails;
 
        ///////////////////////////////////////////////////////////
        //
        //      FUNCTION DEFINITIONS
        //
 
        function showInternalSavingsDetails() {
            dialog.popin('/states/savingCapacity/internalSavings.popin.html', 'InternalSavingsCtrl', {
                internalScheduledOperations: savingCapacityCtrl.savingCapacity.internalScheduledOperations
            });
        }
 
        function validate() {
            savingCapacityBackend.update(savingCapacityCtrl.savingCapacity).then(function () {
                $window.history.back();
            });
        }
 
 
    });
