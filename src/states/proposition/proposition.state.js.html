'use strict';
 
angular.module('bpce.lea')
    .config(function ($stateProvider, STATES) {
        $stateProvider
            .state(STATES.PROPOSITION, {
                url        : 'proposition',
                templateUrl: '/states/proposition/proposition.html',
                controller : 'PropositionCtrl',
                controllerAs: 'propositionCtrl',
                resolve    : {
                    projects: function (projectsBackend) {
                        return projectsBackend.getProposition();
                    }
                }
            })
            .state(STATES.CONFIG_RESERVE, {
                url         : 'configreserve/:externalReference',
                templateUrl : '/states/proposition/configProjet/config.reserve.html',
                controller  : 'ConfigReserveCtrl',
                controllerAs: 'configReserveCtrl',
                resolve     : {
                    proposition: function (projectsBackend, $stateParams) {
                        return projectsBackend.getPropositionDetail($stateParams.externalReference)
                            .then(function (result) {
                                return result;
                            });
                    }
                }
 
            }).state(STATES.CONFIG_PROJECT, {
            url        : 'configproject/:externalReference',
            templateUrl: '/states/proposition/configProjet/config.project.html',
            controller : 'ConfigProjectCtrl',
            resolve    : {
                proposition: function (projectsBackend, $stateParams) {
                    return projectsBackend.getPropositionDetail($stateParams.externalReference)
                        .then(function (result) {
                            return result;
                        });
                }
            }
        });
 
    });
