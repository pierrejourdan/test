'use strict';
 
angular.module('bpce.lea')
    .controller('MainCtrl', function () {
    })
    .run(function ($rootScope, localStorage, $window) {
        $rootScope.$on('$stateChangeStart', function (e, toState, toParams) {
            var referential = localStorage.getReferential();
            if (referential && toParams.p) {
                referential.projectConfiguration[0].label = toParams.p.toUpperCase();
                localStorage.setReferential(referential);
            }
 
            if (referential) {
                $window.jQuery('html').removeClass().addClass(referential.projectConfiguration[0].label + '-styles');
            }
        });
 
    });
