(function () {
    'use strict';
 
    angular.module('bpce.lea.services')
        .factory('localStorage', localStorageFactory);
 
    function localStorageFactory($window, $translate, $rootScope, moment) {
 
        ///////////////////////////////////////////////////////////
        //
        //      INIT
        //
 
        var storage;
        initStorage();
 
        ///////////////////////////////////////////////////////////
        //
        //      EXPORTS
        //
 
        return {
            getConfiguration   : getConfiguration,
            getReferential     : getReferential,
            getRelation        : getRelation,
            getProfile         : getProfile,
            getSession         : getSession,
            getToken           : getToken,
            removeConfiguration: removeConfiguration,
            removeReferential  : removeReferential,
            removeRelation     : removeRelation,
            removeProfile      : removeProfile,
            removeSession      : removeSession,
            removeStorage      : removeStorage,
            setConfiguration   : setConfiguration,
            setReferential     : setReferential,
            setRelation        : setRelation,
            setProfile         : setProfile,
            setSession         : setSession,
            setStorage         : setStorage
        };
 
 
        ///////////////////////////////////////////////////////////
        //
        //      FUNCTION DEFINITIONS
        //
 
        function getRelation() {
            return storage.get('relation');
        }
 
        function setRelation(relation) {
            storage.set('relation', relation);
        }
 
        function removeRelation() {
            storage.del('relation');
        }
 
        function getProfile() {
            return storage.get('profile');
        }
 
        function setProfile(profile) {
            storage.set('profile', profile);
        }
 
        function removeProfile() {
            storage.del('profile');
        }
 
        function getSession() {
            return storage.get('session');
        }
 
        function setSession(session) {
            storage.set('session', session);
        }
 
        function removeSession() {
            storage.del('session');
        }
 
        function setStorage(_token) {
            storage.set('bpce_storage', {
                token : _token,
                loaded: moment().format('X'),
                lang  : $translate.use()
            });
        }
 
        function removeStorage() {
            storage.del('bpce_storage');
        }
 
        function getToken() {
            return validToken();
        }
 
        function getConfiguration() {
            var configuration = storage.get('configuration');
            return configuration && (configuration.expiresIn > moment().format('X')) ? configuration : undefined;
        }
 
        function setConfiguration(configuration) {
            configuration.expiresIn = moment().add(1, 'day').format('X');
            storage.set('configuration', configuration);
        }
 
        function removeConfiguration() {
            storage.del('configuration');
        }
 
        function getReferential() {
            var referential = storage.get('referential');
            return referential && (referential.expiresIn > moment().format('X')) && $translate.use() === referential.lang ? referential : undefined;
        }
 
        function setReferential(referential) {
            referential.lang = $translate.use();
            referential.expiresIn = moment().add(1, 'day').format('X');
            storage.set('referential', referential);
        }
 
        function removeReferential() {
            storage.del('referential');
        }
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    AUXILIARY FUNCTIONS
        //
 
        function initStorage() {
 
            function BrowserStorage(_sessionStorage) {
                this.sessionStorage = _sessionStorage;
            }
 
            BrowserStorage.prototype.set = function (_key, _value) {
                if (_value) {
                    this.sessionStorage.setItem(_key, JSON.stringify(_value));
                }
            };
 
            BrowserStorage.prototype.get = function (_key) {
                var item = this.sessionStorage.getItem(_key);
                if (item) {
                    return JSON.parse(item);
                }
                else {
                    return undefined;
                }
            };
 
            BrowserStorage.prototype.del = function (_key) {
                this.sessionStorage.removeItem(_key);
            };
 
            function MemoryStorage() {
                this.sessionStorage = {};
            }
 
            MemoryStorage.prototype.set = function (_key, _value) {
                if (_value) {
                    this.sessionStorage[_key] = _value;
                }
            };
 
            MemoryStorage.prototype.get = function (_key) {
                return this.sessionStorage[_key];
            };
 
            MemoryStorage.prototype.del = function (_key) {
                this.sessionStorage[_key] = undefined;
            };
 
            try {
                $window.sessionStorage.setItem("test", "");
                $window.sessionStorage.removeItem("test");
                storage = new BrowserStorage($window.sessionStorage);
            }
            catch (err) {
                storage = new MemoryStorage();
            }
        }
 
        function validToken() {
            var _storage = storage.get('bpce_storage');
 
            if (_storage && $translate.use() === _storage.lang && _storage.token && _storage.token.access_token) {
                var today = moment().format('X');
                var tokenDate = parseInt(_storage.loaded) + parseInt(_storage.token.expires_in);
 
                if (today < tokenDate) {
                    return _storage.token;
                }
            }
            if (_storage) { //Token invalid -> logout
                $rootScope.$broadcast('logout');
            }
            return undefined;
        }
    }
})();
