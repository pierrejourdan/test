'use strict';
 
angular.module('bpce.lea.services')
    .factory('dialog', function ($rootScope, ngDialog) {
 
        function myFunction(val) {
            return function () {
                return val;
            };
        }
 
        function transformDataToResolve(datas) {
            if (angular.isObject(datas)) {
                var resolves = {};
                for (var property in datas) {
                    if (datas.hasOwnProperty(property)) {
                        // do stuff
                        var value = datas[property];
                        resolves[property] = myFunction(value);
                    }
                }
                return resolves;
            }
        }
 
        return {
 
            alert: function (message) {
                var $scope = $rootScope.$new();
                $scope.message = message;
 
                return ngDialog.open({
                    template : '<p data-ng-bind-html=\'message | translate:description\'></p>' +
                    '<div class="ngdialog-buttons">' +
                    '<button class="o-btn o-btn--secondary" ng-click="closeThisDialog()" translate="COMMON_OK">' +
                    '</div>',
                    plain    : true,
                    showClose: false,
                    className: 'ngdialog-theme-bpce',
                    scope    : $scope
                });
            },
 
            confirm: function (message, otherConfirmLabel, otherCancelLabel) {
                var $scope = $rootScope.$new();
                $scope.message = message;
 
                var confirmLabel = otherConfirmLabel ? otherConfirmLabel : 'COMMON_CONFIRM';
                var cancelLabel = otherCancelLabel ? otherCancelLabel : 'COMMON_CANCEL';
                return ngDialog.openConfirm({
                    template : '<p>{{message | translate}}</p>' +
                    '<div class="ngdialog-buttons">' +
                    '<button class="o-btn o-btn--secondary ngdialog-button" ng-click="closeThisDialog()" translate="' + cancelLabel + '"></button>' +
                    '<button class="o-btn o-btn--primary ngdialog-button" ng-click="confirm()" translate="' + confirmLabel + '"></button>' +
                    '</div>',
                    plain    : true,
                    showClose: false,
                    className: 'ngdialog-theme-bpce',
                    scope    : $scope
                });
            },
 
            isOpen: function (id) {
                return ngDialog.isOpen(id);
            },
 
            close: function (id, data) {
                ngDialog.close(id, data);
            },
 
            popin: function (url, controller, data, cssclass, showClose) {
 
                var params = {
                    template       : url,
                    id             : "popin",
                    plain          : false,
                    className      : 'ngdialog-theme-bpce-popin',
                    closeByDocument: false,
                    controller     : controller,
                    showClose      : showClose,
                    closeByEscape  : false,
                    resolve        : transformDataToResolve(data)
                };
 
                if (cssclass) {
                    params.className = params.className + " " + cssclass;
                }
 
                return ngDialog.openConfirm(params);
            }
        };
    });
