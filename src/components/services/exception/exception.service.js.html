'use strict';
 
angular.module('bpce.lea.services')
    .factory('exception', function ($log, $translate, dialog) {
 
        return {
            display: display,
            log    : log
        };
 
        function log(message) {
            var _message, _description;
            if (message.error) {
                _message = message.error.error || message.error.message || message.error;
                _description = {description: (message.error.error_description || '')};
            } else {
                _message = message;
            }
 
            _message = $translate.instant(_message, _description, 'description');
 
            $log.error(_message);
 
            return _message;
        }
 
        function display(message) {
            dialog.alert(log(message));
        }
 
    });
