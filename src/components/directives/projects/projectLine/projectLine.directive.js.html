(function () {
    'use strict';
 
    angular.module('bpce.lea')
        .directive('projectLine', projectLine);
 
    function projectLine() {
        return {
            restrict   : 'E',
            replace    : true,
            scope      : {
                project   : '=',
                nbProjects: '=',
                toValidate: '='
            },
            controller : projectLineController,
            templateUrl: '/components/directives/projects/projectLine/projectLine.html'
        };
    }
 
    function projectLineController($scope, $translate, $state, dialog, projectsBackend, localStorage, STATES) {
 
        /////////////////////////////////////////////////////
        //
        //     Bindable elements
        //
 
        $scope.isPropositionState = isPropositionState();
 
        $scope.project.isDefault = isDefaultProject($scope.project);
        $scope.project.isReserve = isReserveProject($scope.project);
        $scope.project.isResidual = isResidualProject($scope.project);
 
        $scope.project.isEditable = isProjectEditable($scope.project);
        $scope.project.isRemovable = isProjectRemovable($scope.project);
 
        $scope.project.isValidated = isValidatedProject($scope.project);
 
        $scope.project.priority = adjustProjectPriority($scope.project);
        $scope.project.allocatedAmount = adjustProjectAllocatedAmount($scope.project);
 
        $scope.removeProject = removeProject;
        $scope.updateProject = updateProject;
        $scope.validateProject = validateProject;
 
 
        /////////////////////////////////////////////////////
        //
        //     State functions definitions
        //
 
        function isPropositionState() {
            return $state.current.name === STATES.PROPOSITION;
        }
 
        function isDefaultProject(project) {
            return isReserveProject(project) || isResidualProject(project);
        }
 
        function isReserveProject(project) {
            return project.type === 'Reserve';
        }
 
        function isResidualProject(project) {
            return project.type === 'Residual';
        }
 
        /*function isAllocatedProject(project) {
         return project.status === 'Allocated';
         }*/
 
        function isNonAllocatedProject(project) {
            return project.status === 'NonAllocated';
        }
 
        function isValidatedProject(project) {
            return project.status === 'Validated';
        }
 
        /////////////////////////////////////////////////////
        //
        //     Function definitions
        //
 
        function adjustProjectAllocatedAmount(project) {
            if (isResidualProject(project) && !project.allocatedAmount) {
                return $translate.instant('PROJECT_NO_RESIDUAL');
            } else if (isNonAllocatedProject(project)) {
                return $translate.instant('PROJECT_NON_ALLOCATED');
            } else {
                return project.allocatedAmount === null ? '-' : project.allocatedAmount;
            }
        }
 
        function adjustProjectPriority(project) {
            return isDefaultProject(project) ? $translate.instant('PROJECT_DEFAULT_PRIORITY') : project.priority;
        }
 
        function isProjectEditable(project) {
            return !isResidualProject(project);
        }
 
        function isProjectRemovable(project) {
            return !isDefaultProject(project);
        }
 
        function removeProject() {
            dialog.confirm('PROJECT_CONFIRM_BEFORE_DELETE').then(function () {
                projectsBackend.remove($scope.project.externalReference).then(function () {
                    $scope.$emit('UPDATE_PROJECT_LIST');
                });
            });
        }
 
        function updateProject() {
            var referential = localStorage.getReferential();
            var category = _.find(referential.objectiveTypes, {key: $scope.project.type});
            dialog.popin('/components/directives/projects/projectPopins/projectCaracteristics.popin.html',
                'ProjectCaracteristicsCtrl',
                {
                    project                : $scope.project,
                    category               : category,
                    mode                   : 'update',
                    priorityOptions        : [{value: 1}, {value: 2}],
                    priorityOptionsSelected: {value: $scope.project.priority},
                    nbProjects             : $scope.nbProjects
                }
            ).then(function () {
                $scope.$emit('UPDATE_PROJECT_LIST');
            });
        }
 
        function validateProject(project) {
            if (project.isValidated) {
                // TODO: unvalidate other projects
                console.log('Goto project proposition again');
            } else {
                if (isReserveProject(project)) {
                    $state.go(STATES.CONFIG_RESERVE, {externalReference: project.externalReference});
                } else {
                    $state.go(STATES.CONFIG_PROJECT, {externalReference: project.externalReference});
                }
            }
        }
 
 
    }
 
})();
