(function () {
    'use strict';
 
    angular.module('bpce.lea.directives')
        .directive('projectList', projectList);
 
    function projectList() {
        return {
            restrict   : 'E',
            replace    : true,
            scope      : {
                projects: '='
            },
            controller : projectListController,
            templateUrl: '/components/directives/projects/projectList/projectList.html'
        };
    }
 
    function projectListController($scope, $state, $filter, $translate, projectsBackend, dialog, localStorage, STATES) {
 
        // Bindable
        $scope.addProject = selectCategory;
        $scope.nbValidatedProjects = nbValidatedProjects;
        $scope.reserveProject = _.find($scope.projects, {type: 'Reserve'});
        $scope.residualProject = _.find($scope.projects, {type: 'Residual'});
        $scope.userProjects = filterUserProjects();
        $scope.$on('UPDATE_PROJECT_LIST', updateProjectsList);
 
 
        function nbValidatedProjects() {
            return _.filter($scope.projects, {status: 'Validated'}).length;
        }
 
        function filterUserProjects() {
            return _.filter($scope.projects, function (project) {
                /*jshint bitwise: false*/
                return !~['Reserve', 'Residual'].indexOf(project.type);
            });
        }
 
        function selectCategory() {
            var referential = localStorage.getReferential();
            dialog.popin('/components/directives/projects/projectPopins/chooseCategory.popin.html',
                'ChooseCategoryCtrl',
                {
                    categories: _.filter(referential.objectiveTypes, {isSelectable: true})
                }
            ).then(function (selectedCategory) {
                addProject(selectedCategory);
            });
        }
 
        function addProject(category) {
            dialog.popin('/components/directives/projects/projectPopins/projectCaracteristics.popin.html',
                'ProjectCaracteristicsCtrl',
                {
                    project   : {
                        priority      : $scope.userProjects.length + 1,
                        type          : category.key,
                        desiredAmount : undefined,
                        horizonInYears: parseInt($filter('referentialFilter')('IAS.Recommendation.Projects.DefaultHorizon.NbYears', 'applConfig'))
                    },
                    category  : category,
                    mode      : 'create',
                    nbProjects: $scope.userProjects.length + 1
                }
            ).then(function (project) {
                if (project) {
                    updateProjectsList();
                    dialog.confirm($translate.instant('ADD_PROJECT_AGAIN'), $translate.instant('COMMON_YES'), $translate.instant('COMMON_NO')).then(function () {
                        selectCategory();
                    });
                }
            }, function (cause) {
                if (cause === 'back') {
                    selectCategory();
                }
 
            });
        }
 
 
        function updateProjectsList() {
            if ($state.current.name === STATES.SITUATION) {
                projectsBackend.getDefault().then(refreshPage);
            } else {
                projectsBackend.getProposition().then(refreshPage);
            }
        }
 
        function refreshPage(data) {
            $scope.projects = data.projects;
            $scope.userProjects = filterUserProjects();
        }
    }
 
})();
