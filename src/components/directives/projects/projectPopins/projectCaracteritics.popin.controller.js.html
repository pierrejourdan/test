'use strict';
 
angular.module('bpce.lea')
    .controller('ProjectCaracteristicsCtrl', function ($scope, mode, project, projectsBackend, category, rfc4122, nbProjects) {
 
        $scope.mode = mode;
        $scope.project = angular.copy(project);
        $scope.category = category;
 
        $scope.validate = validate;
        $scope.cancel = cancel;
        $scope.backToCategories = backToCategories;
        $scope.priorityOptions = getPriorityOptions(nbProjects);
        $scope.prioritySelected = {value: project.priority};
 
        function getPriorityOptions(nbProjects) {
            var priorities = [];
            for (var i = 0; i < nbProjects; i++) {
                priorities.push({value: i + 1});
            }
            return priorities;
        }
 
        function backToCategories() {
            $scope.closeThisDialog('back');
        }
 
        function validate() {
            switch ($scope.mode) {
                case 'update':
                    var project = {
                        type             : $scope.project.type,
                        name             : $scope.project.name,
                        desiredAmount    : $scope.project.desiredAmount,
                        date             : moment().add($scope.project.horizonInYears, 'y').format('YYYY-MM-DD'),
                        priority         : $scope.prioritySelected.value,
                        externalReference: $scope.project.externalReference
                    };
 
                    projectsBackend.update(project).then(function (updatedProject) {
                        $scope.confirm(updatedProject);
                    });
                    break;
                case 'create':
                    // On génère le guid pour le projet.
                    $scope.project.externalReference = rfc4122.v4();
                    // On convertit le nb d'années en date, par rapport à la date du jour.
                    $scope.project.date = moment().add($scope.project.horizonInYears, 'y').format('YYYY-MM-DD');
                    $scope.project.priority = $scope.prioritySelected.value;
                    projectsBackend.add($scope.project).then(function () {
                        $scope.confirm($scope.project);
                    });
                    break;
                default:
                    $scope.closeThisDialog();
                    break;
            }
        }
 
        function cancel() {
            $scope.closeThisDialog();
        }
 
    });
