(function () {
    'use strict';
 
    angular.module('bpce.lea')
        .directive('projectionChart', projectionChart);
 
    function projectionChart($window) {
        return {
            restrict   : 'E',
            scope      : {
                data           : '=',
                objectiveYear  : '=',
                objectiveAmount: '='
            },
            replace    : true,
            templateUrl: '/components/directives/projectionChart/projectionChart.html',
            link       : function (scope, element) {
 
                // HTML/SVG ELEMENTS
                var tooltip, tooltipWidth;
                var pastillesContainer, pastilleFavourable, pastilleNeutral, pastilleUnfavourable;
                var axisContainer, axisX, axisY;
                var chartContainer, outerPath, innerPath, medianPath;
                var objectiveContainer, objectiveLineH, objectiveLineV, objectivePoint, objectiveLabel;
                var overlayContainer;
 
                var dataTransformer = {};
                var scale = {};
 
                var svgWidth, svgHeight, chartWidth, chartHeight;
                var margin = {top: 5, right: 20, bottom: 20, left: 50};
 
                scope.selectedData = {
                    date            : undefined,
                    veryFavourable  : undefined,
                    favourable      : undefined,
                    neutre          : undefined,
                    unfavourable    : undefined,
                    veryUnfavourable: undefined
                };
 
                function init() {
 
                    svgWidth = element.parent().width();
                    svgHeight = angular.element('.d-projection-chart-container').height();
                    chartWidth = svgWidth - margin.left - margin.right;
                    chartHeight = svgHeight - margin.top - margin.bottom;
 
                    var svg = d3.select('svg');
                    svg.select('.d-projection-chart-svg').attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');
                    tooltip = d3.select('#tooltip');
                    tooltipWidth = angular.element('#tooltip')[0].offsetWidth;
 
                    pastillesContainer = svg.select('#pointsContainer');
                    pastilleFavourable = pastillesContainer.select('#indicatorFavourable').attr('cy', chartHeight).attr('rx', 4).attr('ry', 4);
                    pastilleNeutral = pastillesContainer.select('#indicatorNeutral').attr('cy', chartHeight).attr('rx', 4).attr('ry', 4);
                    pastilleUnfavourable = pastillesContainer.select('#indicatorUnfavourable').attr('cy', chartHeight).attr('rx', 4).attr('ry', 4);
 
                    axisContainer = svg.select('#axisContainer');
                    axisX = axisContainer.select('#xAxis');
                    axisY = axisContainer.select('#yAxis');
 
                    chartContainer = svg.select('#chartContainer');
                    outerPath = chartContainer.select('#outerPath');
                    innerPath = chartContainer.select('#innerPath');
                    medianPath = chartContainer.select('#medianPath');
 
                    objectiveContainer = svg.select('#objectiveContainer');
                    objectiveLineH = svg.select('#objectiveH');
                    objectiveLineV = svg.select('#objectiveV');
                    objectivePoint = svg.select('#objectivePoint').attr('rx', 4).attr('ry', 4);
                    objectiveLabel = svg.select('#objectiveLabel');
 
                    overlayContainer = svg.select('#overlayContainer');
 
 
                    dataTransformer.outerArea = d3.area()
                        .curve(d3.curveCardinal.tension(0.5))
                        .x(function (d) {
                            return scale.x(d.date);
                        })
                        .y0(function (d) {
                            return scale.y(d.veryFavourable);
                        })
                        .y1(function (d) {
                            return scale.y(d.veryUnfavourable);
                        });
 
                    dataTransformer.innerArea = d3.area()
                        .curve(d3.curveCardinal.tension(0.5))
                        .x(function (d) {
                            return scale.x(d.date);
                        })
                        .y0(function (d) {
                            return scale.y(d.vafourable);
                        })
                        .y1(function (d) {
                            return scale.y(d.unvafourable);
                        });
 
                    dataTransformer.medianLine = d3.line()
                        .curve(d3.curveCardinal.tension(0.5))
                        .x(function (d) {
                            return scale.x(d.date);
                        })
                        .y(function (d) {
                            return scale.y(d.neutral);
                        });
                }
 
                function drawAxis(chartWidth, chartHeight) {
 
                    var xAxis = d3.axisBottom(scale.x)
                        .tickSizeInner(-chartHeight).tickSizeOuter(0).tickPadding(10);
 
                    var yAxis = d3.axisLeft(scale.y)
                        .tickSizeInner(-chartWidth).tickSizeOuter(0).tickPadding(5);
 
                    axisX.attr('transform', 'translate(0, ' + chartHeight + ')').call(xAxis);
                    axisY.call(yAxis);
                }
 
                function drawOverlays(data) {
                    // bind data to overlays
                    var overlays = overlayContainer.selectAll('.overlay')
                        .data(data);
 
                    //remove old overlays
                    overlays.exit()
                        .on('mouseover', null)
                        .on('mouseleave', null)
                        .remove();
 
                    //add new overlays
                    var allOverlays = overlays.enter().append('rect')
                        .on('mouseover', showTooltip)
                        .on('mouseleave', hideTooltip)
                        .merge(overlays);
 
                    // style all overlays
                    allOverlays.attr('class', 'overlay')
                        .attr('x', function (d, i) {
                            if (i === 0) {
                                return scale.x(d.date) - scale.x(data[i + 1].date) / 2;
                            } else {
                                return scale.x(d.date) - (scale.x(d.date) - scale.x(data[i - 1].date)) / 2;
                            }
                        })
                        .attr('width', function (d, i) {
                            if (i === 0) {
                                return scale.x(data[i + 1].date);
                            } else {
                                return scale.x(d.date) - scale.x(data[i - 1].date);
                            }
                        })
                        .attr('height', chartHeight);
 
 
                }
 
                function hideTooltip() {
                    tooltip.style('opacity', '0');
                    pastillesContainer.style('opacity', '0');
                }
 
                function showTooltip(d) {
                    var posX = scale.x(d.date);
                    tooltip.style('opacity', '1');
                    var tooltipPos = Math.min(chartWidth + 5 + margin.right - posX, svgWidth - tooltipWidth);
                    tooltip.style('right', tooltipPos + 'px');
 
                    pastillesContainer.style('opacity', '1');
                    pastilleFavourable.transition().duration(100).attr('cx', posX).attr('cy', scale.y(d.vafourable));
                    pastilleNeutral.transition().duration(100).attr('cx', posX).attr('cy', scale.y(d.neutral));
                    pastilleUnfavourable.transition().duration(100).attr('cx', posX).attr('cy', scale.y(d.unvafourable));
 
 
                    scope.$apply(function () {
 
                        scope.selectedData.date = d.date.getFullYear();
                        scope.selectedData.veryFavourable = d.veryFavourable;
                        scope.selectedData.favourable = d.vafourable;
                        scope.selectedData.neutre = d.neutral;
                        scope.selectedData.unfavourable = d.unvafourable;
                        scope.selectedData.veryUnfavourable = d.veryUnfavourable;
 
                    });
 
                }
 
                function drawPaths(data, animate) {
                    if (animate) {
                        outerPath.datum(data).transition()
                            .attr('d', dataTransformer.outerArea);
 
                        innerPath.datum(data).transition()
                            .attr('d', dataTransformer.innerArea);
 
                        medianPath.datum(data).transition()
                            .attr('d', dataTransformer.medianLine);
                    } else {
                        outerPath.datum(data)
                            .attr('d', dataTransformer.outerArea);
 
                        innerPath.datum(data)
                            .attr('d', dataTransformer.innerArea);
 
                        medianPath.datum(data)
                            .attr('d', dataTransformer.medianLine);
 
                    }
                }
 
                function drawObjective() {
                    var objectiveDate = d3.timeParse("%Y")(scope.objectiveYear);
 
                    objectiveLineH.transition().duration(100).attr('x1', 0)
                        .attr('x2', scale.x(objectiveDate))
                        .attr('y1', scale.y(scope.objectiveAmount))
                        .attr('y2', scale.y(scope.objectiveAmount));
 
                    objectiveLineV.transition().duration(100).attr('x1', scale.x(objectiveDate))
                        .attr('x2', scale.x(objectiveDate))
                        .attr('y1', chartHeight)
                        .attr('y2', scale.y(scope.objectiveAmount));
 
                    objectivePoint.transition().duration(100).attr('cx', scale.x(objectiveDate))
                        .attr('cy', scale.y(scope.objectiveAmount));
 
                    objectiveLabel.transition().duration(100).attr('transform', 'translate(' + scale.x(objectiveDate) + ',' + scale.y(scope.objectiveAmount) + 50 + ')');
 
                }
 
                function updateChart(animate) {
                    var data = scope.data.map(function (d) {
                        return {
                            date            : d3.timeParse("%Y")(d.year),
                            veryUnfavourable: d.tres_defavorable,
                            unvafourable    : d.defavorable,
                            neutral         : d.objective,
                            vafourable      : d.favorable,
                            veryFavourable  : d.tres_favorable
                        };
                    });
                    scale.x = d3.scaleTime().range([0, chartWidth])
                        .domain(d3.extent(data, function (d) {
                            return d.date;
                        }));
                    scale.y = d3.scaleLinear().range([chartHeight, 0])
                        .domain([d3.min(data, function (d) {
                            return d.veryUnfavourable;
                        }), d3.max(data, function (d) {
                            return d.veryFavourable;
                        })]);
 
 
                    drawAxis(chartWidth, chartHeight);
                    drawPaths(data, animate);
                    drawObjective();
                    drawOverlays(data);
                }
 
 
                init();
 
                scope.$watch('data', function () {
 
 
                    updateChart(true);
 
                });
 
                scope.$watch('objectiveYear', function () {
 
 
                    updateChart(true);
 
                });
 
                scope.$watch('objectiveAmount', function () {
 
 
                    updateChart(true);
 
                });
 
 
                angular.element($window).on('resize', function () {
                    // recompute sizes
                    svgWidth = element.parent().width();
                    chartWidth = svgWidth - margin.left - margin.right;
 
                    //update chart without anims
                    updateChart(false);
                });
 
                scope.$on('$destroy', function () {
                    angular.element($window).off('resize');
                });
            }
        };
    }
 
})();
