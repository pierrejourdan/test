'use strict';
 
angular.module('bpce.lea.directives').directive('pieChart', function ($window) {
        return {
            replace    : true,
            restrict   : 'E',
            scope      : {
                data: '='
            },
            templateUrl: '/components/directives/pieChart/pieChart.html',
            link       : function (scope, element) {
 
                /////////////////
                // VARIABLES
                /////////////////
                var pie, pieContainer, tooltip, svg, fixedTooltip, radius, arc;
 
                var height = 165;
                var valueMin = 10;
                var valueSuffix = '%';
                var tooltipPaddingLeft = 15;
 
                var width = element.parent().width();
 
 
                /////////////////////
                // FUNCTIONS
                /////////////////////
                function init() {
                    radius = Math.min(width - 10, height - 10) / 2;
 
                    // DEFINE PIE & FIXED TOOLTIP
                    pie = d3.pie().value(function (d) {
                            return d.count;
                        })
                        .sort(null);
 
                    fixedTooltip = d3.tip()
                        .attr("class", "d-pie-chart__fixed-tooltip")
                        .offset([(radius / 2) + 15, 0])
                        .html(function (d) {
                            return d.data.count + valueSuffix;
                        });
 
                    // CREATE SVG / PIE CONTAINER / TOOLTIP & LEGENDS CONTAINER
                    svg = d3.select('#pie-chart').select('svg').attr('width', width).attr('height', height).call(fixedTooltip);
                    pieContainer = svg.select('g.d-pie-chart__pie-container').attr('transform', 'translate(' + (width / 2) + ',' + (height / 2) + ')');
                    tooltip = d3.select('.d-pie-chart__tooltip-container');
                }
 
                function addSlices() {
                    var slicesSelection = pieContainer.selectAll('.d-pie-chart__slice--container')
                        .data(function () {
                            return pie(scope.data);
                        });
 
                    var slicesG = slicesSelection
                        .enter()
                        .append('g')
                        .attr('class', 'd-pie-chart__slice--container')
                        .merge(slicesSelection);
 
                    slicesSelection.exit().remove();
 
                    return slicesG;
                }
 
                function addSlicesPath(slicesG, color) {
                    arc = d3.arc().innerRadius(0).outerRadius(radius);
 
                    var slicesPathSelection = slicesG.selectAll('.d-pie-chart__slice-path')
                        .data(function (d) {
                            return [d];
                        });
 
                    var slicesPath = slicesPathSelection
                        .enter()
                        .append('path')
                        .attr('class', 'd-pie-chart__slice-path')
                        .merge(slicesPathSelection)
                        .attr('d', arc)
                        .attr('fill', function (d) {
                            return color(d.data.label);
                        })
                        ;
                    slicesPath.transition().duration(500)
                        .attrTween("d", function (d) {
                            this._current = this._current || d;
                            var interpolate = d3.interpolate(this._current, d);
                            this._current = interpolate(0);
                            return function (t) {
                                return arc(interpolate(t));
                            };
                        });
                    slicesPathSelection.exit().remove();
 
                    return slicesPath;
                }
 
                function addSlicesLabel(slicesG) {
                    var labelArc = d3.arc()
                        .outerRadius(radius - 1 / 3 * radius)
                        .innerRadius(radius - 1 / 3 * radius);
 
                    var slicesTextSelection = slicesG.selectAll('.d-pie-chart__slice-label')
                        .data(function (d) {
                            return [d];
                        });
 
                    slicesTextSelection
                        .enter()
                        .append('text')
                        .attr("class", "d-pie-chart__slice-label")
                        .attr("text-anchor", "middle")
                        .attr("fill", "white")
                        .merge(slicesTextSelection)
                        .transition()
                        .duration(400)
                        .attr("transform", function (d) {
                            return "translate(" + labelArc.centroid(d) + ")";
                        })
                        .text(function (d) {
                            if (d.data.count > valueMin) {
                                return d.data.count + valueSuffix;
                            }
                        });
 
                    slicesTextSelection.exit().remove();
                }
 
                function addEventsOnSlicesPath(slicesPath) {
                    slicesPath
                        .on("mouseleave", function () {
                            unhighlight(d3.select(this));
                            hideDynamicTooltip();
                            closeAllSlices(slicesPath);
                        })
                        .on('mousemove', function (d) {
                            showDynamicTooltip(d);
                        })
                        .on('mouseenter', function () {
                            highlight(d3.select(this));
                            closeAllSlices(slicesPath);
                            openSlice(d3.select(this));
                        });
                }
 
                function addEventsOnLegends(legendsItem, slicesPath) {
                    legendsItem
                        .on("mouseleave", function (d, i) {
                            var associatedSlice = getAssociatedSlice(slicesPath, i);
                            if (associatedSlice.selection) {
                                unhighlight(associatedSlice.selection);
                                closeAllSlices(slicesPath);
                            }
                            hideDynamicTooltip();
                            hideFixedTooltip();
                        })
                        .on('mouseenter', function (d, i) {
                            var associatedSlice = getAssociatedSlice(slicesPath, i);
                            if (associatedSlice.selection) {
                                highlight(associatedSlice.selection);
                                closeAllSlices(slicesPath);
                                openSlice(associatedSlice.selection);
                                showFixedTooltip(associatedSlice.data, associatedSlice.element);
                            }
                        });
                }
 
                function getAssociatedSlice(slicesPath, i) {
                    var associatedSlice = {};
                    slicesPath.each(function (d2) {
                        if (i === d2.index) {
                            associatedSlice.selection = d3.select(this);
                            associatedSlice.data = d2;
                            associatedSlice.element = slicesPath._groups[d2.index][0];
                        }
                    });
                    return associatedSlice;
                }
 
                function closeAllSlices(slices) {
                    slices.transition().duration(400).attr('transform', 'translate(0,0)').on('end', function (d) {
                        d.data.isSelected = false;
                    });
                }
 
                function openSlice(slice) {
                    slice.transition()
                        .duration(400)
                        .attr("transform", function (d) {
                            if (!d.data.isSelected) {
                                var c = arc.centroid(d),
                                    x = c[0],
                                    y = c[1],
                                    h = Math.sqrt(x * x + y * y),
                                    pullOutSize = parseInt(5, 10);
 
                                return "translate(" + ((x / h) * pullOutSize) + ',' + ((y / h) * pullOutSize) + ")";
                            }
                            else {
                                return 'translate(0,0)';
                            }
                        })
                        .on('end', function (d) {
                            d.data.isSelected = !d.data.isSelected;
                        });
                }
 
                function highlight(slice) {
                    slice.style('opacity', 0.7);
                }
 
                function unhighlight(slice) {
                    slice.style('opacity', 1);
                }
 
                function showDynamicTooltip(d) {
                    if (d.data.count < valueMin) {
                        tooltip.style("left", (d3.event.pageX + tooltipPaddingLeft) + "px")
                            .style("top", d3.event.pageY + "px");
                        tooltip.select('.d-pie-chart__tooltip-count').html(d.data.count + '%');
                        tooltip.style('display', 'block');
                    }
                }
 
                function hideDynamicTooltip() {
                    tooltip.style('display', 'none');
                }
 
                function showFixedTooltip(d, slice) {
                    if (d.data.count < valueMin) {
                        fixedTooltip.show(d, slice);
                    }
                }
 
                function hideFixedTooltip() {
                    fixedTooltip.hide();
                }
 
 
                function updatePieChart() {
                    var color = d3.scaleOrdinal().range(_.map(scope.data, 'color'));
 
                    var slicesG = addSlices();
                    var slicesPath = addSlicesPath(slicesG, color);
                    addSlicesLabel(slicesG, color);
                    addEventsOnSlicesPath(slicesPath);
                    addEventsOnLegends(d3.selectAll('.d-pie-chart__legend-item'), slicesPath);
                }
 
                init();
 
                /////////////////////
                // ON LOAD DATA
                /////////////////////
                var unregisterDataWatcher = scope.$watch('data', function () {
                    updatePieChart();
                });
 
                angular.element($window).on('resize', function () {
                    width = element.parent().width();
                    init();
                    updatePieChart();
                });
 
                scope.$on('$destroy', function () {
                    unregisterDataWatcher();
                    angular.element($window).off('resize');
                });
            }
        };
    }
);
