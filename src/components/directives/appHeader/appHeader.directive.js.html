'use strict';
 
angular.module('bpce.lea.directives')
    .directive('appHeader', function (profileCompletenessBackend) {
        return {
            templateUrl     : '/components/directives/appHeader/appHeader.html',
            restrict        : 'E',
            replace         : true,
            controllerAs    : 'headerCtrl',
            bindToController: true,
            controller      : function (localStorage) {
                var headerCtrl = this;
                var referential = localStorage.getReferential();
 
                // Attributes
                headerCtrl.profileCompleteness = localStorage.getProfile();
                headerCtrl.showMenu = false;
                headerCtrl.profiles = referential.profiles;
                headerCtrl.kAndEProfiles = referential.kAndEProfiles;
 
                // Functions
                headerCtrl.selectRiskProfile = selectRiskProfile;
                headerCtrl.toggleProfileCompleteness = toggleProfileCompleteness;
                headerCtrl.profileComplete = profileComplete;
 
                function toggleProfileCompleteness() {
                    headerCtrl.showMenu = !headerCtrl.showMenu;
                }
 
                function profileComplete() {
                    return headerCtrl.profileCompleteness.savingCapacity && headerCtrl.profileCompleteness.externalAsset && headerCtrl.profileCompleteness.riskProfile.isComplete && headerCtrl.profileCompleteness.kAndE.isComplete;
                }
 
                function selectRiskProfile(riskProfile) {
                    profileCompletenessBackend.put(riskProfile.key)
                        .then(function () {
                            //toggleProfileCompleteness();
                            headerCtrl.profileCompleteness.riskProfile.isComplete = true;
                            headerCtrl.profileCompleteness.riskProfile.creationDate = moment().format('YYYY-MM-DDThh:mm:ss');
                            headerCtrl.profileCompleteness.riskProfile.value = riskProfile.key;
                        });
                }
            }
        };
    });
