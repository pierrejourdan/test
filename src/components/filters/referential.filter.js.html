'use strict';
 
angular.module('bpce.lea.filters')
    .filter("referentialFilter", function (localStorage) {
 
        return function (key, domain, specificAttribute) {
            if (!key) {
                return '';
            }
 
            var referential = localStorage.getReferential() || {},
                ref = _.find(referential[domain], function (elt) {
                    return elt.key === key.toString();
                });
 
            return ref ? (specificAttribute ? ref[specificAttribute] : ref.label ) : key;
        };
 
    });
