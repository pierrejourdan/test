(function () {
    'use strict';
 
    angular.module('bpce.lea.backend')
        .factory('contextBackend', contextBackend);
 
    function contextBackend($q, localStorage, backend, heartBeat, Uri, rfc4122) {
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    EXPORTS
        //
 
        return {
            referenceData: referenceData,
            context      : context,
            createSession: createSession
        };
 
        function context() {
            var deferred = $q.defer();
 
            backend.get(Uri.BASE + '/greetings/context')
                .then(function (contextInfo) {
                    localStorage.setConfiguration(contextInfo);
                    if (contextInfo.sessionKeepAliveUrl) {
                        heartBeat.start();
                    }
                    deferred.resolve(contextInfo);
                }, function (error) {
                    deferred.reject(error);
                });
 
            return deferred.promise;
        }
 
        function referenceData() {
            var deferred = $q.defer();
 
            backend.get(Uri.BASE + '/data/reference_data')
                .then(function (referential) {
                    localStorage.setReferential(referential);
                    deferred.resolve(referential);
                }, function (error) {
                    deferred.reject(error);
                });
 
            return deferred.promise;
        }
 
        function createSession() {
            return backend.post(Uri.BASE + '/advice/advicesequence/?externalId=' + rfc4122.v4())
                .then(storeSession);
        }
 
        function storeSession(session) {
            localStorage.setSession(session);
        }
 
 
    }
 
})();
