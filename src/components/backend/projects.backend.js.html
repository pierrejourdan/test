(function () {
    'use strict';
 
    angular.module('bpce.lea.backend')
        .factory('projectsBackend', projectsBackend);
 
    function projectsBackend(localStorage, Uri, backend) {
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    EXPORTS
        //
 
        return {
            getDefault               : getDefault,
            getProposition           : getProposition,
            getPropositionDetail     : getPropositionDetail,
            simulatePropositionDetail: simulatePropositionDetail,
            remove                   : remove,
            add                      : add,
            update                   : update,
        };
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    FUNCTION DEFINITIONS
        //
 
        function getSessionId() {
            return localStorage.getSession().id;
        }
 
        function getDefault() {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/computedefaultprojects';
            return backend.post(url);
        }
 
        function getProposition() {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/computeprojects';
            return backend.post(url);
        }
 
        function getPropositionDetail(externalReference) {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/projects/' + externalReference + '/projectDetailAllocation';
            return backend.get(url);
        }
 
        function simulatePropositionDetail(externalReference, data) {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/projects/' + externalReference + '/projectDetailAllocation';
            return backend.post(url, data);
        }
 
        function remove(externalReference) {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/projects/' + externalReference;
            return backend.remove(url);
        }
 
        function add(project) {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/projects';
            return backend.post(url, project);
        }
 
        function update(project) {
            var url = Uri.BASE + '/advice/advicesequence/' + getSessionId() + '/projects/' + project.externalReference;
            return backend.put(url, project);
        }
    }
 
})();
