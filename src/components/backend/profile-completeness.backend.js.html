(function () {
    'use strict';
 
    angular.module('bpce.lea.backend')
        .factory('profileCompletenessBackend', profileCompletenessBackend);
 
    function profileCompletenessBackend(backend, localStorage) {
 
        ///////////////////////////////////////////////////////////
        //
        //      EXPORTS
        //
 
        return {
            get: get,
            put: put,
            log: log
        };
 
        ///////////////////////////////////////////////////////////
        //
        //      FUNCTION DEFINITIONS
        //
 
        function get() {
            var adviceSequenceId = localStorage.getSession().id;
            return backend.get(backend.baseURL + '/advice/advicesequence/' + adviceSequenceId + '/Completeness');
        }
 
        function put(profileRiskId) {
            var adviceSequenceId = localStorage.getSession().id;
            return backend.put(backend.baseURL + '/advice/advicesequence/' + adviceSequenceId + '/riskProfile', '"'+profileRiskId+'"');
        }
 
        function log() {
            //api/v1/advice/advicesequence/{adviceSequenceId}/recommendations/
            var adviceSequenceId = localStorage.getSession().id;
            return backend.post(backend.baseURL + '/advice/advicesequence/' + adviceSequenceId + '/recommendations');
        }
 
    }
})();
