(function () {
    'use strict';
 
    angular.module('bpce.lea.backend')
        .factory('loginBackend', loginBackend);
 
    function loginBackend(backend) {
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    CONFIG
        //
 
        var authenticationBaseURL = '/api/auth';
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    EXPORTS
        //
 
        return {
            login        : login,
            externalLogin: externalLogin,
            logout       : logout
        };
 
        function login(identifier, password) {
            return authenticationBaseURLPost('/token', $.param({
                grant_type: 'password',
                Username  : identifier,
                Password  : password
            }));
        }
 
        function externalLogin(params) {
            params.grant_type = 'sjr';
            return authenticationBaseURLPost('/token', $.param(_.omitBy(params, _.isNil)));
        }
 
        function logout() {
            return authenticationBaseURLPost(authenticationBaseURL + '/logout');
        }
 
 
        ////////////////////////////////////////////////////////////////////////////////////////////
        //
        //    AUXILIARY FUNCTIONS
        //
 
        function authenticationBaseURLPost(url, params) {
            return backend.post(authenticationBaseURL + url, params);
        }
    }
})();
