(function () {
    'use strict';
 
    angular.module('bpce.lea.backend')
        .factory('savingCapacityBackend', savingCapacityBackend);
 
    function savingCapacityBackend(backend, Uri, localStorage) {
 
        ///////////////////////////////////////////////////////////
        //
        //      EXPORTS
        //
 
        return {
            get   : get,
            update: update
        };
 
        ///////////////////////////////////////////////////////////
        //
        //      FUNCTION DEFINITIONS
        //
 
        function get() {
            var adviceSequenceId = localStorage.getSession().id;
            return backend.get(Uri.BASE + '/advice/advicesequence/' + adviceSequenceId + '/saving_capacity');
        }
 
        function update(savingCapacity) {
            var adviceSequenceId = localStorage.getSession().id;
            return backend.put(Uri.BASE + '/advice/advicesequence/' + adviceSequenceId + '/saving_capacity', savingCapacity);
        }
    }
})();
